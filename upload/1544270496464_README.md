```js
//单个
t.load("share",function(){
  new wshare(cfg)
})
//多个组件
t.load(['share','register','tbxCalculator'],function(){
  new wshare(cfg)      //分享
  new wregister(cfg) //注册
  new tbxCalculator(cfg)  //铜本息计算器
})
```
